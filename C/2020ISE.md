Code to check if a number is a palindrome:
#include<stdio.h>

int main()
{
    int a,b,c,sum;
    printf("Enter the number:\t");
    scanf("%d",&a);
    sum=0;
    c=a;
    while(a>0)
    {
        b=a%10;
        sum=sum*10+b;
        a=a/10;
    }
    if(sum==c)
    {
        printf("The number %d is a palindrome",c);
    }
    else
    {
        printf("The number %d is not a palindrome",c);
    }
    return 0;
}
